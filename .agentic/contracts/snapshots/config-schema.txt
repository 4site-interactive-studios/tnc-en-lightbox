interface ENLightboxConfig {
  header?: string | undefined
  body?: string | undefined
  image?: ENLightboxImage | undefined
  cta?: ENLightboxCta | undefined
  secondaryCta?: ENLightboxSecondaryCta | undefined
  dismissLabel?: string | undefined
  closeOnOverlay?: boolean | undefined
  closeOnEsc?: boolean | undefined
  hideImageOnMobile?: boolean | undefined
  triggers?: TriggersConfigBase | undefined
  theme?: ThemeConfigBase | undefined
  layout?: LayoutConfigBase | undefined
}

interface NormalizedConfig {
  header: string  // default src.header ?? ''
  body: string  // default src.body ?? ''
  image?: ENLightboxImage | undefined  // default src.image
  cta?: ENLightboxCta | undefined  // default src.cta
  secondaryCta?: ENLightboxSecondaryCta | undefined  // default src.secondaryCta
  dismissLabel?: string | undefined  // default src.dismissLabel
  closeOnOverlay: boolean  // default src.closeOnOverlay ?? true
  closeOnEsc: boolean  // default src.closeOnEsc ?? true
  hideImageOnMobile: boolean  // default topLevelHideImageOnMobile
  triggers: TriggersConfigBase  // default src.triggers ?? {}
  theme: NormalizedTheme  // default normalizeTheme(src.theme)
  layout: NormalizedLayout  // default normalizeLayout(src.layout, topLevelHideImageOnMobile)
}

// Resolved layout shape from src/themes/config.ts
interface NormalizedLayout {
  variant: "two-column"
  imagePosition: LayoutImagePosition
  imageRatio: string
  hideImageOnMobile?: boolean | undefined
  closeButton: LayoutCloseButton
}
